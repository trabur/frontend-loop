Distributed Programming

(with dataflow graphs)

the paradigm shift here is that we are coding the flow of data rather than the flow of control or logic. this level of abstraction and thinking requires that we make decisions about what Nodes and Edges are needed as dependancies. because of that each one of these graphs consists of an implementation and an interface. our interface is a simple library coded in JavaScript which describes the what and how of data flow through Agents and Subgraphs. we leave the final implementation up to the programmer.

function agent(flow) {
  flow.import('input', (message) => {
    console.log(`welcome, ${message}`)
  })
}
let node = fl.node('initializing')
node.subgraph('subgraph', subgraph)
node.agent('agent', agent)
node.subgraph('subgraph').port('output').port('input').agent('agent')
// --or--
// subgraph(${name_of_subgraph})

// subgraph() output -> input agent()
function clone(flow) {
  flow.import('input', (a) => {
    flow.exports('clone', a, a, a, a)
  })
}
let node = fl.node('first')
node.in('db_path').port('input').agent('clone', clone)
node.agent('clone').ports('clone', '0').out('db_path0')
node.agent('clone').ports('clone', '1').out('db_path1')
node.agent('clone').ports('clone', '2').out('db_path2')
node.agent('clone').ports('clone', '3').out('db_path3')
// --or--
// db_path => input clone(${msg_clone})
// clone() clone[0] => db_path0
// clone() clone[1] => db_path1
// clone() clone[2] => db_path2
// clone() clone[3] => db_path3
function adder(flow) {
  flow.imports('add', (a, b, c, d) => {
    let value = a + b + c + d
    flow.export('output', value)
  })
}
let node = fl.node('second')
node.in('add0').ports('add', '0').agent('adder', adder)
node.in('add1').ports('add', '1').agent('adder')
node.in('add2').ports('add', '2').agent('adder')
node.in('add3').ports('add', '3').agent('adder').port('output').out('output')
// --or--
// add0 => add[0] adder(${path_to_adder})
// add1 => add[1] adder()
// add2 => add[2] adder()
// add3 => add[3] adder() output => output
function nand(flow) {
  flow.imports('nand', (a, b) => {
    if (a && b) {
      flow.export('output', false)
    } else {
      flow.export('output', true)
    }
  })
}
let node = fl.node('third')
node.in('input').port('input').agent('clone', clone)
node.agent('clone').ports('clone', '0').ports('nand', 'a').agent('nand', nand)
node.agent('clone').ports('clone', '1').ports('nand', 'b').agent('nand').port('output').out('output')
// --or--
// input => input clone(${msg_clone})
// clone() clone[0] -> a nand(${maths_boolean_nand})
// clone() clone[1] -> b nand() output => output
